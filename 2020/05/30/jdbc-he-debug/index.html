JDBC

介绍

JDBC API是一个Java API，可以访问任何类型表列数据，特别是存储在关系数据库中的数据。JDBC代表Java数据库连接。相当于一个中间商。

步骤

1、加载驱动

1.1、导包

导入相关的数据库jar包

1.2、 加载驱动

这一步相当于你准备打开了手机

Class.forName(yourDriverName);

2、获得连接

这一步相当于你打开了美团外卖

Connection connection = DriverManager.getConnection(url, username, password);

3、书写sql

这一步相当于你点了要吃的东西

String sql="select * from student";// 查找

4、准备发起对话

相当于你下单了，外卖小哥接到了你的单子

PreparedStatement ps = connection.prepareStatement(sql);

5、执行操作并接收结果

相当于外卖小哥去饭馆买了饭给你送过来，你要去拿

/*
情况一：执行查询操作的时候  要调用的方法是executeQuery()，其返回值是ResultSet类型
*/
ResultSet rs = ps.executeQuery();
// 这时候就需要遍历rs，进行下一步操作（直接输出或者放到对象里边返回）
while(rs.next()) {
    // 这里进行业务代码的书写
	System.out.print(rs.getString("Sname") + "   ");
	System.out.println(rs.getInt("Sage"));
}

/*
情况二：执行删除，增加，修改操作的时候，调用方法execute()或者executeUpdate()
区别:executeUpdate() 返回的是受影响的行数
	execute() 执行成功返回false
*/
boolean execute = ps.execute();// 返回执行成功了没有
// 或者
int executeUpdate = ps.executeUpdate();// 返回执行的行数

改进

我们发现用上面的方法写业务的时候会出现大量代码冗余的情况，所以我们自己封装一下，可以减少很多冗余

1、书写配置文件

driver=com.mysql.cj.jdbc.Driver #驱动名
jdbcUrl=jdbc:mysql://localhost:3306/test?serverTimezone=GMT%2B8 #固定格式，只需要修改地址和数据库名以及后面的参数
username=root #用户名
userpassword=root #密码

2、创建JdbcUtil类

package com.baidu.util;
import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.sql.Statement;
import java.util.ResourceBundle;

/*
 * jdbc工具类
 * @author Zhy
 *
 */
public class JDBCUtil {
	private static String driver;
	private static String jdbcUrl;
	private static String username;
	private static String userpassword;
	static{
		//读取Properties文件
		ResourceBundle bundle = ResourceBundle.getBundle("jdbc");
		driver = bundle.getString("driver");
		jdbcUrl= bundle.getString("jdbcUrl");
		username = bundle.getString("username");
		userpassword = bundle.getString("userpassword");
		try {
			Class.forName(driver);
		} catch (ClassNotFoundException e) {
			e.printStackTrace();
		}
	}
	//获取Connection对象
	public static Connection getConnection(){
		Connection conn = null;
		try {
		 conn =	DriverManager.getConnection(jdbcUrl, username, userpassword);
		} catch (SQLException e) {
			e.printStackTrace();
		}
		return conn;		
	}
	//关闭Statement
	public static void closeStatement(Statement state){
			try {
				if(state != null){
					state.close();
				}
			} catch (SQLException e) {
				e.printStackTrace();
			}
	}
	//关闭Connection
	public static void closeConnection(Connection conn){
		try {
			if(conn != null){
				conn.close();
			}
		} catch (SQLException e) {
			e.printStackTrace();
		}
	}
	//关闭资源
	public static void closeResource(Statement state,Connection conn,ResultSet rs){
		closeStatement(state);
		closeConnection(conn);
		closeResultSet(rs);
	}
	//关闭ResultSet连接
	public static void closeResultSet(ResultSet rs){
		try {
			if(rs != null){
				rs.close();
			}
		} catch (SQLException e) {
			e.printStackTrace();
		}
	}
	//事务回滚
	public static void rollback(Connection conn){
		try {
			if(conn != null){
				conn.rollback();
			}
		} catch (SQLException e) {
			e.printStackTrace();
		}
	}
	public static void commit(Connection conn) {
		try {
			if(conn!=null) {
				conn.commit();
			}
		} catch (Exception e) {			
			e.printStackTrace();
		}
	}
	/*
	 * 测试连接Connection
	 */
	public static void main(String[] args) {
		Connection con = null;
		con = JDBCUtil.getConnection();
		System.out.println(con);
	}
}

3、书写代码

一个简单的例子：

public static void main(String[] args) {
		// 调用工具类获取连接
		Connection connection = JDBCUtil.getConnection();
    	//书写sql
		String sql = "select * from school limit 0,10";
		try {
            //准备开始对话
			PreparedStatement prepareStatement = connection.prepareStatement(sql);
            //接收结果
			ResultSet rs = prepareStatement.executeQuery();	
            //遍历结果集，写业务代码
			while(rs.next()) {
                //这里写业务代码
				System.out.println(rs.getString("name"));
			}		
		} catch (SQLException e) {
			e.printStackTrace();
		}	
	}
